from typing import Optional
import pytest
from .say_my_name import say_my_name


def test_say_my_name():
    # Correctly passing first_name and last_name
    assert say_my_name("John","Smith") == "My name is John Smith"
    assert say_my_name("Luis") == "My name is Luis"
    
    # None passed as last_name
    with pytest.raises(TypeError):
        say_my_name("John", None)
        
    # None passed as first_name
    with pytest.raises(TypeError):
        say_my_name(None)
        
    # Number passed as first_name
    with pytest.raises(TypeError):
        say_my_name(1)
        
    # Number passed as last_name
    with pytest.raises(TypeError):
        say_my_name("John", 0)
        
    # Missing two arguments
    with pytest.raises(TypeError):
        say_my_name()
